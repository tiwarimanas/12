import pandas as pd df1 = pd.DataFrame({'ID':[1,2], 'Name':['A','B']}) df2 = pd.DataFrame({'ID':[1,2], 'Sal':[100,200]}) df3 = pd.DataFrame({'ID':[3], 'Name':['C']}) # Merge (like SQL Join) merged = pd.merge(df1, df2, on='ID') # Concatenation (stacking) concat = pd.concat([df1, df3], ignore_index=True) print("Merged:\n", merged) print("Concatenated:\n", concat)
Terminal Output

Unit 5: Database Queries using SQL

Creating Database and Table
mysql> CREATE DATABASE school; Query OK, 1 row affected (0.01 sec) mysql> USE school; Database changed mysql> CREATE TABLE student ( -> id INT PRIMARY KEY, -> name VARCHAR(50), -> marks INT, -> stream VARCHAR(20) -> ); Query OK, 0 rows affected (0.02 sec)
INSERT, UPDATE, DELETE Operations
mysql> INSERT INTO student VALUES (101, 'Amit', 85, 'Science'); Query OK, 1 row affected (0.01 sec) mysql> UPDATE student SET marks = 90 WHERE id = 101; Query OK, 1 row affected (0.01 sec) mysql> DELETE FROM student WHERE marks < 33; Query OK, 0 rows affected (0.00 sec)
Math Functions: POWER, ROUND, MOD
mysql> SELECT POWER(2,3), ROUND(15.678, 1), MOD(10,3); +------------+-----------------+----------+ | POWER(2,3) | ROUND(15.678,1) | MOD(10,3)| +------------+-----------------+----------+ | 8 | 15.7 | 1 | +------------+-----------------+----------+ 1 row in set (0.00 sec)
Text Functions: UPPER, LOWER
mysql> SELECT UCASE('hello'), LOWER('WORLD'); +----------------+----------------+ | UCASE('hello') | LOWER('WORLD') | +----------------+----------------+ | HELLO | world | +----------------+----------------+ 1 row in set (0.00 sec)
Text Functions: SUBSTR, LENGTH
mysql> SELECT SUBSTR('Informatics', 3, 4), LENGTH('Data'); +-----------------------------+----------------+ | SUBSTR('Informatics', 3, 4) | LENGTH('Data') | +-----------------------------+----------------+ | form | 4 | +-----------------------------+----------------+ 1 row in set (0.00 sec)
Text Functions: LEFT, RIGHT, INSTR
mysql> SELECT LEFT('India', 2), RIGHT('India', 2), INSTR('banana', 'a'); +------------------+-------------------+----------------------+ | LEFT('India', 2) | RIGHT('India', 2) | INSTR('banana', 'a') | +------------------+-------------------+----------------------+ | In | ia | 2 | +------------------+-------------------+----------------------+ 1 row in set (0.00 sec)
Text Functions: TRIM Operations
mysql> SELECT LENGTH(TRIM(' ABC ')); +---------------------------+ | LENGTH(TRIM(' ABC ')) | +---------------------------+ | 3 | +---------------------------+ 1 row in set (0.00 sec)
Date Functions
mysql> SELECT NOW(), YEAR(NOW()), DAYNAME(NOW()); +---------------------+-------------+----------------+ | NOW() | YEAR(NOW()) | DAYNAME(NOW()) | +---------------------+-------------+----------------+ | 2023-10-24 10:00:00 | 2023 | Tuesday | +---------------------+-------------+----------------+ 1 row in set (0.00 sec)
Aggregate Functions
mysql> SELECT MAX(marks), MIN(marks), AVG(marks) FROM student; +------------+------------+------------+ | MAX(marks) | MIN(marks) | AVG(marks) | +------------+------------+------------+ | 98 | 65 | 82.5000 | +------------+------------+------------+ 1 row in set (0.01 sec)
GROUP BY Clause
mysql> SELECT stream, COUNT(*) FROM student GROUP BY stream; +----------+----------+ | stream | COUNT(*) | +----------+----------+ | Science | 15 | | Commerce | 12 | +----------+----------+ 2 rows in set (0.01 sec)
HAVING Clause
mysql> SELECT stream, AVG(marks) FROM student -> GROUP BY stream -> HAVING AVG(marks) > 80; +---------+------------+ | stream | AVG(marks) | +---------+------------+ | Science | 85.5000 | +---------+------------+ 1 row in set (0.01 sec)
ORDER BY Clause
mysql> SELECT name, marks FROM student ORDER BY marks DESC; +-------+-------+ | name | marks | +-------+-------+ | Rohan | 98 | | Amit | 90 | | Jia | 85 | +-------+-------+ 3 rows in set (0.00 sec)
WHERE Clause Conditions (IN, BETWEEN, LIKE)
mysql> SELECT * FROM student WHERE marks BETWEEN 80 AND 90; mysql> SELECT * FROM student WHERE name LIKE 'A%'; mysql> SELECT * FROM student WHERE stream IN ('Science', 'Arts'); Query OK (0.00 sec)
DISTINCT and LIMIT
mysql> SELECT DISTINCT stream FROM student; +----------+ | stream | +----------+ | Science | | Commerce | +----------+ 2 rows in set (0.01 sec) mysql> SELECT * FROM student LIMIT 2; +-----+------+-------+---------+ | id | name | marks | stream | +-----+------+-------+---------+ | 101 | Amit | 90 | Science | | 102 | Ria | 92 | Science | +-----+------+-------+---------+
Equi-Join (Inner Join)
mysql> SELECT student.name, fees.amount -> FROM student, fees -> WHERE student.id = fees.id; +------+--------+ | name | amount | +------+--------+ | Amit | 5000 | | Ria | 5500 | +------+--------+ 2 rows in set (0.02 sec)
